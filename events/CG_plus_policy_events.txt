country_event={
	id=CG_plus_policy_uniform.1000
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		OR ={
		has_country_flag = carnalism_flag
		has_country_flag = asceticism_flag
		}	
		is_ai = no
	}

	immediate = {
		if = { # 派送 修道女服 学园事件
			limit = {has_policy_flag =  nun_uniform }
			random_list = {
				1={modifier = {factor = 0 has_country_flag = asceticism_flag } country_event = {id = CG_plus_policy_uniform.1100 }}
				1={country_event = {id = CG_plus_policy_uniform.1101 }}
				1={country_event = {id = CG_plus_policy_uniform.1102 }}
				1={country_event = {id = CG_plus_policy_uniform.1103 }}
				1={country_event = {id = CG_plus_policy_uniform.1104 }}
				1={modifier = {factor = 0 has_country_flag = asceticism_flag } country_event = {id = CG_plus_policy_uniform.1105 }}
			

		}}
		if = { # 派送 公主礼装 学园事件
			limit = {has_policy_flag =  princess_uniform }
			random_list = {
				1={country_event = {id = CG_plus_policy_uniform.1200 }}
				1={ modifier = {factor = 0 has_country_flag = asceticism_flag } country_event = {id = CG_plus_policy_uniform.1201 }}
				1={country_event = {id = CG_plus_policy_uniform.1202 }}

				1={ modifier = {factor = 0 has_country_flag = asceticism_flag } country_event = {id = CG_plus_policy_uniform.1201 }}
				1={modifier = {factor = 0 has_country_flag = carnalism_flag } country_event = {id = CG_plus_policy_uniform.1204 }}
		}}
		if = { # 派送 群星军服 学园事件
			limit = {has_policy_flag =  army_uniform }
			random_list = {
				1={country_event = {id = CG_plus_policy_uniform.1300 }}

				1={country_event = {id = CG_plus_policy_uniform.1303 }}
				1={country_event = {id = CG_plus_policy_uniform.1304 }}
				1={ modifier = {factor = 0 has_country_flag = asceticism_flag } country_event = {id = CG_plus_policy_uniform.1305 }}
				

		}}
		if = { # 派送 衬衣中裙 学园事件
			limit = {has_policy_flag =  school_uniform has_technology = tech_beauty_theory }
			random_list = {
				1={country_event = {id = CG_plus_policy_uniform.1100 }}
		}}
		if = { # 派送 短水手服 学园事件
			limit = {has_policy_flag =  sailorfuku_uniform }
			random_list = {
				1={modifier = {factor = 0 has_country_flag = asceticism_flag } country_event = {id = CG_plus_policy_uniform.1500 }}
				1={country_event = {id = CG_plus_policy_uniform.1501 }}
				1={modifier = {factor = 0 has_country_flag = carnalism_flag } country_event = {id = CG_plus_policy_uniform.1502 }}
				1={modifier = {factor = 0 has_country_flag = asceticism_flag }country_event = {id = CG_plus_policy_uniform.1503 }}
				1={country_event = {id = CG_plus_policy_uniform.1504 }}
				1={modifier = {factor = 0 has_country_flag = asceticism_flag }country_event = {id = CG_plus_policy_uniform.1505 }}
		}}
		if = { # 派送 艳兔女郎 学园事件
			limit = {has_policy_flag =  bunny_uniform }
			random_list = {
				1={modifier = {factor = 0 has_country_flag = carnalism_flag }country_event = {id = CG_plus_policy_uniform.1600 }}
				1={country_event = {id = CG_plus_policy_uniform.1601 }}
				1={country_event = {id = CG_plus_policy_uniform.1602 }}
				1={modifier = {factor = 0 has_country_flag = asceticism_flag }country_event = {id = CG_plus_policy_uniform.1603 }}
				1={country_event = {id = CG_plus_policy_uniform.1604 }}
				1={modifier = {factor = 0 has_country_flag = asceticism_flag }country_event = {id = CG_plus_policy_uniform.1605 }}
		}}
		if = { # 派送 微比坚尼 学园事件
			limit = {has_policy_flag =  microbikini_uniform }
			random_list = {
				1={country_event = {id = CG_plus_policy_uniform.1700} }
